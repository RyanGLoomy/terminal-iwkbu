ALGORITMA MODUL SISTEM
Sistem Integrasi Terminal IWKBU

1) Modul Registrasi PO
PROCEDURE RegistrasiPO: INPUT nama_perusahaan, email, password IF email = "" OR password = "" THEN SHOW "Email dan password wajib diisi" ELSE user = AUTH_SIGNUP(email, password, metadata{role:"po"}) IF user NOT EXISTS THEN SHOW "Registrasi gagal, email sudah terdaftar" ELSE DB_INSERT("INSERT INTO po (id, nama_perusahaan, email, status_verifikasi) VALUES (?, ?, ?, 'menunggu')", user.id, nama_perusahaan, email) LOG "Registrasi PO berhasil, menunggu verifikasi" SHOW "Registrasi berhasil, menunggu verifikasi dari Staf IW" END IF END IF END PROCEDURE

2) Modul Login
PROCEDURE Login: INPUT email, password IF email = "" OR password = "" THEN SHOW "Email dan password wajib diisi" ELSE IF RATE_LIMIT_EXCEEDED(email, ip) THEN SHOW "Terlalu banyak percobaan, coba lagi nanti" ELSE user = AUTH_SIGNIN(email, password) IF user NOT EXISTS THEN LOG "LOGIN_GAGAL: kredensial salah" SHOW "Email atau password salah" ELSE profile = DB_QUERY("SELECT is_active FROM profiles WHERE id = ?", user.id) IF profile.is_active = FALSE THEN AUTH_SIGNOUT() SHOW "Akun tidak aktif, hubungi admin" ELSE role = RESOLVE_ROLE(user.id) IF role NOT FOUND THEN SHOW "Role tidak ditemukan, hubungi admin" ELSE SET SESSION user_id = user.id SET SESSION role = role LOG "LOGIN: berhasil sebagai " + role CLEAR_RATE_LIMIT(email, ip) SET COOKIE session GOTO Dashboard(role) END IF END IF END IF END IF END IF END PROCEDURE

3) Modul Logout
PROCEDURE Logout: INPUT konfirmasi_logout IF konfirmasi_logout = TRUE THEN AUTH_SIGNOUT() CLEAR COOKIE session LOG "LOGOUT: user keluar" GOTO Login ELSE SHOW "Logout dibatalkan" END IF END PROCEDURE

4) Modul Kelola Profil
PROCEDURE KelolaProfil: INPUT full_name profile = DB_QUERY("SELECT full_name FROM profiles WHERE id = ?", SESSION.user_id) SHOW profile.full_name INPUT full_name_baru IF full_name_baru = "" THEN SHOW "Nama tidak boleh kosong" ELSE DB_UPDATE("UPDATE profiles SET full_name = ? WHERE id = ?", full_name_baru, SESSION.user_id) LOG "UPDATE_USER: profil diperbarui" SHOW "Profil berhasil diperbarui" END IF END PROCEDURE

5) Modul Manajemen Akun
PROCEDURE ManajemenAkun: IF SESSION.role = "admin-terminal" THEN daftar_petugas = DB_QUERY("SELECT id, nama, terminal_id FROM petugas_terminal WHERE terminal_id = ?", SESSION.terminal_id) SHOW daftar_petugas INPUT nama, email, password IF nama = "" THEN SHOW "Nama wajib diisi" ELSE user = AUTH_ADMIN_CREATEUSER(email, password) DB_INSERT("INSERT INTO profiles (id, email) VALUES (?, ?)", user.id, email) DB_INSERT("INSERT INTO user_roles (user_id, role_id) VALUES (?, 'loket')", user.id) DB_INSERT("INSERT INTO petugas_terminal (nama, terminal_id) VALUES (?, ?)", nama, SESSION.terminal_id) LOG "BUAT_USER: petugas loket dibuat" SHOW "Petugas berhasil dibuat" END IF ELSE IF SESSION.role = "staf-iw" THEN daftar_user = DB_QUERY("SELECT p.id, p.email, p.full_name, r.name FROM profiles p JOIN user_roles ur ON p.id = ur.user_id JOIN roles r ON ur.role_id = r.id") SHOW daftar_user INPUT user_id, role_baru DB_DELETE("DELETE FROM user_roles WHERE user_id = ?", user_id) DB_INSERT("INSERT INTO user_roles (user_id, role_id) VALUES (?, ?)", user_id, role_baru) LOG "UPDATE_USER: role diperbarui" SHOW "Role pengguna berhasil diperbarui" END IF END IF END PROCEDURE

6) Modul Manajemen Data Armada
PROCEDURE ManajemenDataArmada: IF aksi = "update_status" THEN INPUT armada_id, status_baru armada = DB_QUERY("SELECT id FROM armada WHERE id = ? AND po_id = ?", armada_id, SESSION.po_id) IF armada NOT EXISTS THEN SHOW "Armada tidak ditemukan" ELSE DB_UPDATE("UPDATE armada SET status_operasional = ? WHERE id = ?", status_baru, armada_id) LOG "UPDATE_ARMADA: status diperbarui" SHOW "Status armada diperbarui" END IF ELSE IF aksi = "import_iwkbu" THEN INPUT file_upload plates = PARSE_FILE(file_upload) FOR EACH plate IN plates normalized = NORMALIZE_PLATE(plate) DB_UPSERT("INSERT INTO iwkbu_source_records (nomor_polisi, compliance_status) VALUES (?, ?) ON CONFLICT UPDATE", normalized, plate.compliance_status) NEXT plate LOG "IMPORT_IWKBU: " + COUNT(plates) + " baris" SHOW "Import berhasil, " + COUNT(plates) + " baris" END IF END IF END PROCEDURE

7) Modul Buka Sesi Petugas
PROCEDURE BukaSesiPetugas: INPUT pin_input IF RATE_LIMIT_EXCEEDED(user_id) THEN SHOW "Terlalu banyak percobaan PIN" ELSE petugas_list = DB_QUERY("SELECT id, nama, pin_hash FROM petugas_terminal WHERE terminal_id = ? AND is_active = true", SESSION.terminal_id) matched = NULL FOR EACH petugas IN petugas_list IF BCRYPT_COMPARE(pin_input, petugas.pin_hash) = TRUE THEN matched = petugas BREAK END IF NEXT petugas IF matched = NULL THEN LOG "PIN_GAGAL: PIN tidak valid" SHOW "PIN tidak valid" ELSE DB_UPSERT("INSERT INTO petugas_pin_sessions (user_id, petugas_terminal_id, expires_at) VALUES (?, ?, NOW() + 8h) ON CONFLICT UPDATE", SESSION.user_id, matched.id) LOG "VERIFIKASI_PIN: " + matched.nama sesi_aktif = DB_QUERY("SELECT id FROM sesi_petugas WHERE petugas_id = ? AND status = 'aktif'", SESSION.user_id) IF sesi_aktif EXISTS THEN SHOW "Sesi sudah aktif" ELSE DB_INSERT("INSERT INTO sesi_petugas (petugas_id, terminal_id, status) VALUES (?, ?, 'aktif')", SESSION.user_id, SESSION.terminal_id) LOG "BUKA_SESI: sesi dibuka" SHOW "Sesi berhasil dibuka, menu pencatatan tersedia" END IF END IF END IF END PROCEDURE

8) Modul Pencatatan Kendaraan Masuk/Keluar
PROCEDURE PencatatanKendaraan: INPUT nomor_polisi IF SESSION NOT VALID THEN SHOW "Sesi tidak aktif" ELSE IF PIN_SESSION_EXPIRED THEN SHOW "PIN sesi telah berakhir, verifikasi ulang" ELSE armada = DB_QUERY("SELECT id, po_id FROM armada WHERE nomor_polisi ILIKE ? AND status_verifikasi = 'terverifikasi'", nomor_polisi) IF armada NOT EXISTS THEN SHOW "Armada tidak ditemukan atau belum terverifikasi" ELSE DB_INSERT("INSERT INTO kendaraan_masuk (sesi_id, petugas_id, armada_id, po_id, nomor_polisi, waktu_masuk) VALUES (?, ?, ?, ?, ?, NOW())", SESSION.sesi_id, SESSION.user_id, armada.id, armada.po_id, nomor_polisi) LOG "INPUT_TRANSAKSI: kendaraan masuk " + nomor_polisi SHOW "Kendaraan " + nomor_polisi + " berhasil dicatat" END IF END IF END IF END PROCEDURE

9) Modul Rekap dan Riwayat Terminal
PROCEDURE RekapRiwayatTerminal: INPUT filter_tanggal, filter_plat IF SESSION.role = "admin-terminal" THEN rekap = DB_RPC("get_admin_rekap_harian", terminal_id) SHOW rekap ELSE IF SESSION.role = "loket" THEN rekap = DB_RPC("get_admin_rekap_harian", terminal_id, sesi_id) SHOW rekap END IF IF filter_tanggal NOT NULL OR filter_plat NOT NULL THEN rekap = DB_QUERY("SELECT * FROM kendaraan_masuk WHERE waktu_masuk BETWEEN ? AND ? AND nomor_polisi LIKE ?", filter_tanggal_mulai, filter_tanggal_akhir, filter_plat) SHOW rekap END IF END PROCEDURE

10) Modul Kelola Master Data
PROCEDURE KelolaMasterData: daftar_terminal = DB_QUERY("SELECT id, kode, nama FROM terminals") SHOW daftar_terminal INPUT aksi, terminal_id IF aksi = "tambah" THEN INPUT kode, nama DB_INSERT("INSERT INTO terminals (kode, nama) VALUES (?, ?)", kode, nama) LOG "CREATE_TERMINAL: " + nama SHOW "Terminal ditambahkan" ELSE IF aksi = "edit" THEN INPUT kode_baru, nama_baru DB_UPDATE("UPDATE terminals SET kode = ?, nama = ? WHERE id = ?", kode_baru, nama_baru, terminal_id) LOG "UPDATE_TERMINAL: " + nama_baru SHOW "Terminal diperbarui" ELSE IF aksi = "hapus" THEN DB_DELETE("DELETE FROM terminals WHERE id = ?", terminal_id) LOG "DELETE_TERMINAL: " + terminal_id SHOW "Terminal dihapus" END IF END IF END IF END PROCEDURE

11) Modul Rekonsiliasi Data
PROCEDURE RekonsiliasiData: INPUT trigger = "manual" DB_INSERT("INSERT INTO iwkbu_sync_runs (trigger_type, status, started_at) VALUES (?, 'running', NOW())", trigger) run_id = LAST_INSERT_ID() plates = DB_QUERY("SELECT nomor_polisi FROM armada WHERE status_verifikasi = 'terverifikasi'") iwkbu_data = IWKBU_API_POST("/compliance/check", plates) FOR EACH plate IN iwkbu_data DB_UPSERT("INSERT INTO iwkbu_source_records (nomor_polisi, compliance_status, issue_count) VALUES (?, ?, ?) ON CONFLICT UPDATE", plate.nomor_polisi, plate.status, plate.issues) NEXT plate armada_data = DB_QUERY("SELECT a.id, a.po_id, a.nomor_polisi, a.status_verifikasi, a.status_operasional, p.status_verifikasi as po_status FROM armada a JOIN po p ON a.po_id = p.id") FOR EACH armada IN armada_data source = FIND_MATCH(iwkbu_data, armada.nomor_polisi) status = COMPARE(armada, source) DB_UPSERT("INSERT INTO iwkbu_sync_status (armada_id, po_id, reconciliation_status, discrepancy_note) VALUES (?, ?, ?, ?) ON CONFLICT UPDATE", armada.id, armada.po_id, status.reconciliation, status.discrepancy) IF status.reconciliation != "ready" THEN DB_INSERT("INSERT INTO findings (po_id, armada_id, judul, severity, source_type) VALUES (?, ?, ?, ?, 'rekonsiliasi')", armada.po_id, armada.id, "Diskrepansi IWKBU: " + armada.nomor_polisi, status.severity) END IF NEXT armada DB_UPDATE("UPDATE iwkbu_sync_runs SET status = 'success', summary = ?, finished_at = NOW() WHERE id = ?", BUILD_SUMMARY(), run_id) LOG "REKONSILIASI: selesai" SHOW "Rekonsiliasi selesai" END PROCEDURE

12) Modul Temuan dan Klarifikasi
PROCEDURE TemuanDanKlarifikasi: INPUT finding_id, pesan_klarifikasi, file_bukti finding = DB_QUERY("SELECT id, status FROM findings WHERE id = ? AND po_id = ?", finding_id, SESSION.po_id) IF finding NOT EXISTS THEN SHOW "Temuan tidak ditemukan" ELSE IF file_bukti NOT NULL THEN file_path = STORAGE_UPLOAD(file_bukti) END IF DB_INSERT("INSERT INTO finding_clarifications (finding_id, responder_id, message, evidence) VALUES (?, ?, ?, ?)", finding_id, SESSION.user_id, pesan_klarifikasi, file_path) DB_UPDATE("UPDATE findings SET status = 'on_progress' WHERE id = ?", finding_id) LOG "KIRIM_KLARIFIKASI: temuan " + finding_id SHOW "Klarifikasi berhasil dikirim" END IF END PROCEDURE

13) Modul Tindak Lanjut Temuan
PROCEDURE TindakLanjutTemuan: daftar_temuan = DB_QUERY("SELECT f.*, p.nama_perusahaan, a.nomor_polisi FROM findings f JOIN po p ON f.po_id = p.id JOIN armada a ON f.armada_id = a.id WHERE f.status = 'open'") SHOW daftar_temuan INPUT finding_id, keputusan, catatan_resolusi IF keputusan = "terima" THEN DB_UPDATE("UPDATE findings SET status = 'closed', resolved_by = ?, resolution_note = ?, resolved_at = NOW() WHERE id = ?", SESSION.user_id, catatan_resolusi, finding_id) DB_INSERT("INSERT INTO notifications (user_id, title, message) VALUES (?, 'Temuan Ditutup', ?)", po_id, "Temuan Anda telah diterima dan ditutup") LOG "UPDATE_TEMUAN: temuan diterima dan ditutup" SHOW "Temuan diterima dan ditutup" ELSE IF keputusan = "tolak" THEN DB_UPDATE("UPDATE findings SET status = 'closed', resolved_by = ?, resolution_note = ? WHERE id = ?", SESSION.user_id, catatan_resolusi, finding_id) DB_INSERT("INSERT INTO notifications (user_id, title, message) VALUES (?, 'Pelanggaran Valid', ?)", po_id, "Klarifikasi ditolak, pelanggaran dinyatakan valid") LOG "UPDATE_TEMUAN: temuan ditolak, pelanggaran valid" SHOW "Temuan ditolak, pelanggaran dinyatakan valid" END IF END IF END PROCEDURE

14) Modul Laporan, Ekspor, dan Audit Trail
PROCEDURE LaporanEksporAudit: IF aksi = "laporan" THEN INPUT tanggal_mulai, tanggal_akhir, terminal_id rekap = DB_RPC("get_admin_rekap_harian", tanggal_mulai, tanggal_akhir, terminal_id) SHOW rekap INPUT format_ekspor IF format_ekspor = "excel" THEN file = GENERATE_XLSX(rekap) ELSE file = GENERATE_PDF(rekap) END IF DOWNLOAD file ELSE IF aksi = "audit_trail" THEN INPUT filter_aksi, filter_tanggal, kata_kunci logs = DB_RPC("get_activity_logs", filter_aksi, filter_tanggal, kata_kunci) SHOW logs END IF END IF END PROCEDURE
